  * Classe Scanner
    * Obter um valor inteiro:
    Scanner leitor = new Scanner(System.in);
    String x = leitor.nextInteger();
    
    
    * Obter um valor decimal
    Scanner leitor = new Scanner(System.in);
    String x = leitor.nextDouble();
    
    
    * Obter um valor de texto 
    Scanner leitor = new Scanner(System.in);
    String x = leitor.nextString();
